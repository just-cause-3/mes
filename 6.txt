AREA SMALLEST, CODE, READONLY
ENTRY                 ; Mark first instruction to execute

START
    MOV R5, #6            ; Initialize counter to 6 (N = 7)

    LDR R1, =VALUE1       ; Load address of first value
    LDR R2, [R1], #4      ; Load first array element

LOOP
    LDR R4, [R1], #4      ; Load next array element

    CMP R2, R4            ; Compare numbers
    BLS LOOP1             ; If R2 < R4, skip update

    MOV R2, R4            ; Move smaller value into R2

LOOP1
    SUBS R5, R5, #1       ; Decrement counter
    CMP R5, #0
    BNE LOOP              ; Repeat until array ends

    LDR R4, =RESULT       ; Load address of result
    STR R2, [R4]          ; Store smallest number

    NOP
    NOP
    NOP

; Array of 32-bit numbers
VALUE1
    DCD 0X44444444
    DCD 0X22222222
    DCD 0X11111111
    DCD 0X33333333
    DCD 0XAAAAAAAA
    DCD 0X88888888
    DCD 0X99999999

AREA DATA2, DATA, READWRITE
RESULT DCD 0X0

END

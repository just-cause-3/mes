AREA FACTORIAL, CODE, READONLY
ENTRY                 ; Mark first instruction to execute

START
    MOV R0, #7        ; Store factorial number in R0
    MOV R1, R0        ; Copy number to R1

FACT
    SUBS R1, R1, #1   ; Decrement R1
    MUL R3, R0, R1    ; Multiply R0 and R1
    MOV R0, R3        ; Store result back in R0

    CMP R1, #1        ; Compare R1 with 1
    BNE FACT          ; Repeat until R1 = 1

    NOP
    NOP
    NOP

END                   ; End of program

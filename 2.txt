AREA SUM, CODE, READONLY
ENTRY

MOV R1, #10        ; Load 10 to register
MOV R2, #0         ; Clear register to store result

LOOP
    ADD R2, R2, R1 ; Add R1 value to R2
    SUBS R1, #0x01 ; Decrement R1 by 1
    BNE LOOP       ; Repeat until R1 becomes 0

BACK
    B BACK         ; Infinite loop

END
